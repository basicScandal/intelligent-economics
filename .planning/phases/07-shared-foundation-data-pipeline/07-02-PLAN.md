interface MINDScoresFile {
  metadata: {
    generatedAt: string;
    worldBankLastUpdated: string;
    indicatorCount: number;
    countryCount: number;
    version: string;
  };
  indicators: Record<string, {
    name: string;
    dimension: 'm' | 'i' | 'n' | 'd';
    normBounds: { min: number; max: number; p1: number; p99: number };
  }>;
  countries: Record<string, {
    name: string;
    region: string;
    incomeLevel: string;
    dimensions: {
      m: { score: number; indicators: IndicatorValue[]; count: number };
      i: { score: number; indicators: IndicatorValue[]; count: number };
      n: { score: number; indicators: IndicatorValue[]; count: number };
      d: { score: number; indicators: IndicatorValue[]; count: number };
    };
    mind: number;
    bindingConstraint: 'm' | 'i' | 'n' | 'd';
    completeness: number;
  }>;
}

interface IndicatorValue {
  code: string;
  raw: number | null;
  normalized: number | null;
  year: string;
  source: string;
}

    **B. Implement BOM-safe fetch helper:**
    - Use Node.js native `fetch()` (no external HTTP library)
    - Read response as `ArrayBuffer`, decode with `TextDecoder('utf-8')`, strip BOM `\uFEFF`
    - Parse JSON after BOM removal
    - URL pattern: `https://api.worldbank.org/v2/country/all/indicator/{CODE}?format=json&mrnev=1&per_page=300`

    **C. Implement serial fetching with rate limiting:**
    - First fetch country list from `https://api.worldbank.org/v2/country?format=json&per_page=300`
    - Filter to 217 actual countries: `region.id !== 'NA'`
    - Build a `Set<string>` of valid ISO3 country codes
    - Fetch each of the 16 indicators sequentially (NOT Promise.all)
    - Wait 2500ms between each request (`await new Promise(r => setTimeout(r, 2500))`)
    - Add single retry with 10000ms backoff on failure (check if response contains `<html>` or is not valid JSON)
    - Log progress: `console.log('Fetching [N/16] CODE...')`

    **D. Implement normalization pipeline:**
    - Import `normalize` from `../src/lib/mind-score`
    - For each indicator, collect all non-null values across countries
    - Filter OUT aggregate entries (only process ISO3 codes in the valid country Set)
    - Call `normalize(values)` to get percentile-capped bounds
    - Apply the returned `normalize()` function to each country's raw value
    - Store both raw and normalized values per country per indicator

    **E. Implement dimension aggregation (per CONTEXT.md locked decisions):**
    - For each country, for each dimension (m/i/n/d): compute arithmetic mean of available normalized indicator values
    - Require minimum 2 of 4 indicators present per dimension — if fewer, set dimension score to `null` (insufficient data)
    - Track `count` of available indicators per dimension
    - Compute `completeness` as fraction of 16 total indicators with non-null data

    **F. Implement overall MIND score:**
    - Import `calcScore` and `getBindingConstraint` from `../src/lib/mind-score`
    - For countries with all 4 dimension scores: `calcScore({m: dimM, i: dimI, n: dimN, d: dimD})`
    - For countries missing any dimension: set `mind` to `null` and `bindingConstraint` to the dimension with the lowest non-null score (or null if all missing)

    **G. Write output JSON to `src/data/mind-scores.json`:**
    - Use `JSON.stringify(output, null, 2)` for readable committed JSON
    - Create `src/data/` directory if it does not exist (`fs.mkdirSync recursive`)
    - Include `metadata` block: `generatedAt` (ISO date), `worldBankLastUpdated` (from API response header), `indicatorCount` (16), `countryCount` (count of countries in output), `version` ("1.0.0")
    - Include `indicators` block: keyed by indicator code, with name, dimension, and normBounds (min, max, p1, p99)
    - Include `countries` block: keyed by ISO3 code, with full structure per interface

    **H. Error handling:**
    - Wrap entire pipeline in try/catch
    - If any indicator fails after retry, log warning and continue (graceful degradation)
    - If country list fetch fails, abort with clear error message
    - Exit with code 0 on success, code 1 on critical failure

    **I. Console output summary:**
    - After completion, print: total countries, indicators fetched, countries with complete data, countries with incomplete data, total runtime
