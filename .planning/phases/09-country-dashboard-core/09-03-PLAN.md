    **src/scripts/dashboard/state.ts** — Simple state management for dashboard selections:

    ```typescript
    import type { SlimCountry } from './search';

    export interface DashboardState {
      primary: SlimCountry | null;       // Selected country for radar + detail view
      comparison: SlimCountry[];          // Up to 4 countries for bar chart comparison (includes primary)
      isMobile: boolean;                  // Tracks viewport < 768px
    }

    type Listener = (state: DashboardState) => void;

    export function createDashboardState() {
      let state: DashboardState = { primary: null, comparison: [], isMobile: false };
      const listeners: Listener[] = [];

      return {
        get: () => state,
        selectPrimary(country: SlimCountry) { ... },    // Sets primary, adds to comparison if not present (max 4)
        addToComparison(country: SlimCountry) { ... },   // Adds to comparison array (max 4), no-op if already in
        removeFromComparison(code: string) { ... },      // Removes from comparison by code
        setMobile(isMobile: boolean) { ... },            // Updates mobile flag
        subscribe(fn: Listener) { listeners.push(fn); }, // Register state change callback
        notify() { listeners.forEach(fn => fn(state)); }, // Trigger all listeners
      };
    }
    ```

    Rules:
    - `selectPrimary`: sets primary, auto-adds to comparison if not already there (max 4). If comparison already has 4 and country isn't in it, replace the last non-primary entry. Always calls `notify()`.
    - `addToComparison`: appends if comparison.length < 4 and country not already present. Calls `notify()`.
    - `removeFromComparison`: filters out by code. If removed country was primary, set primary to null. Calls `notify()`.
    - `setMobile`: updates isMobile, calls `notify()` only if value changed.

    **src/scripts/dashboard/init.ts** — Main initialization script for the dashboard island.

    IMPORTANT: This script uses TWO separate loading paths (not a single island wrapper):
    - **ECharts loading:** IntersectionObserver on `#dashboard-data` element (client:visible semantics) — dynamically imports echarts-setup.ts and charts.ts when the dashboard area enters the viewport
    - **Search loading:** `requestIdleCallback` (client:idle semantics) — dynamically imports search.ts when the browser is idle

    Structure:

    ```typescript
    // Eagerly import state (tiny, no heavy deps)
    import { createDashboardState } from './state';
    import type { SlimCountry } from './search';
    import type { DimensionKey } from '../../lib/mind-score';

    // 1. Read data eagerly (just JSON parse, no heavy modules)
    const dataEl = document.getElementById('dashboard-data');
    const countries: SlimCountry[] = dataEl ? JSON.parse(dataEl.dataset.scores!) : [];
    const store = createDashboardState();

    // 2. Mobile detection (eager, lightweight)
    store.setMobile(window.matchMedia('(max-width: 767px)').matches);
    window.matchMedia('(max-width: 767px)').addEventListener('change', e => store.setMobile(e.matches));

    // 3. ECharts loading path — IntersectionObserver (client:visible semantics)
    let radarChart: any = null;
    let barChart: any = null;

    const chartObserver = new IntersectionObserver((entries) => {
      if (entries[0].isIntersecting) {
        chartObserver.disconnect();
        // Dynamic import — only loads ECharts when dashboard is visible
        Promise.all([
          import('./echarts-setup'),
          import('./charts'),
        ]).then(([{ echarts }, { makeRadarOption, makeBarOption, getMobileBarOption, getBindingConstraintCallout, siteTheme, DIM_COLORS }]) => {
          radarChart = echarts.init(document.getElementById('radar-chart')!, siteTheme, { renderer: 'svg' });
          barChart = echarts.init(document.getElementById('bar-chart')!, siteTheme, { renderer: 'svg' });

          // ResizeObserver for chart containers
          const ro = new ResizeObserver(() => { radarChart?.resize(); barChart?.resize(); });
          ro.observe(document.getElementById('radar-chart')!);
          ro.observe(document.getElementById('bar-chart')!);

          // Subscribe to state changes for chart rendering
          store.subscribe((state) => {
            // Radar chart
            if (state.primary && state.primary.m !== null) {
              document.getElementById('country-detail')?.classList.remove('hidden');
              radarChart.setOption(makeRadarOption(state.primary));
              radarChart.getDom().setAttribute('aria-label', `MIND dimension radar chart for ${state.primary.name}`);
              // Update binding constraint via DOM IDs
              const bcDimKey = state.primary.bc as DimensionKey;
              const bcScore = state.primary[bcDimKey] ?? 0;
              const bc = getBindingConstraintCallout(bcDimKey, Math.round(bcScore));
              const bcEl = document.getElementById('binding-constraint');
              const bcNameEl = document.getElementById('bc-dimension-name');
              const bcScoreEl = document.getElementById('bc-score');
              const bcTextEl = document.getElementById('bc-text');
              const bcBorderEl = document.getElementById('bc-border');
              if (bcEl) bcEl.classList.remove('hidden');
              if (bcNameEl) bcNameEl.textContent = bc.dimension;
              if (bcScoreEl) bcScoreEl.textContent = `(${Math.round(bcScore)})`;
              if (bcTextEl) bcTextEl.textContent = bc.text;
              if (bcBorderEl) bcBorderEl.style.borderLeftColor = DIM_COLORS[bcDimKey];
            } else {
              document.getElementById('country-detail')?.classList.add('hidden');
            }

            // Bar chart
            if (state.comparison.length > 0) {
              document.getElementById('comparison-section')?.classList.remove('hidden');
              const validCountries = state.comparison.filter(c => c.mind !== null);
              const option = state.isMobile ? getMobileBarOption(validCountries) : makeBarOption(validCountries);
              barChart.setOption(option, true);
              barChart.resize();
              barChart.getDom().setAttribute('aria-label', `MIND dimension comparison for ${validCountries.map(c => c.name).join(', ')}`);
            } else {
              document.getElementById('comparison-section')?.classList.add('hidden');
            }
          });

          // If state already has a primary (search loaded first), render now
          store.notify();
        });
      }
    }, { rootMargin: '200px' });

    if (dataEl) chartObserver.observe(dataEl);

    // 4. Search loading path — requestIdleCallback (client:idle semantics)
    const loadSearch = () => {
      import('./search').then(({ filterCountries, handleSearchKeydown }) => {
        // Build search UI in #search-container
        // ... (implementation details below)
      });
    };

    if ('requestIdleCallback' in window) {
      requestIdleCallback(loadSearch);
    } else {
      setTimeout(loadSearch, 200); // fallback
    }
    ```

    **Search UI implementation** (inside the loadSearch callback):
    - Get `#search-container` element
    - Create `<input type="search" placeholder="Search 217 countries..." />` with Tailwind classes: `w-full px-4 py-3 bg-[#0a0a12] border border-white/10 rounded-lg text-white placeholder-[#8888aa] focus:border-[#00ff88] focus:outline-none text-lg`
    - Create `<div id="search-dropdown" class="hidden absolute ...">` for results dropdown

    - Each dropdown item shows: country name + MIND score (or "Incomplete data" for null). Clicking calls `store.selectPrimary(country)`
    - Dropdown item for countries with mind !== null shows: `<span>{name}</span> <span class="text-[#00ff88]">{mind}</span>`
    - Dropdown item for null-mind countries shows: `<span>{name}</span> <span class="text-[#8888aa]">Incomplete data</span>`
    - Add "Add to comparison" button on each dropdown item (small, secondary style) that calls `store.addToComparison(country)` — only shown for countries with non-null MIND scores

    - Click removes from comparison via `store.removeFromComparison(code)`
    - Chips container is created by init.ts and prepended inside `#comparison-section` before the bar chart

    **Accessibility:**
